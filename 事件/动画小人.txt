    <script>
        var img = document.getElementById('img');
        var w=0,s=0,a=0,d =0;
       document.onkeydown = function(e){
        //    alert(e.keyCode);
        //37  左
        //38  上
        //39  右
        //40  下
        switch(e.keyCode){
            case 37:
                img.style.marginLeft = a * -50 + "px";
                a++;
            break;
            case 38:
                img.style.marginTop= w * -50 + "px";
                w++;
            break;
            case 39:
                img.style.marginLeft = d * 50 + "px";
                d++;
            break;
            case 40:
                img.style.marginTop = s * 50 + "px";
                s++;
            break;
        }
       }
    </script>
